  • Barplot, posição horizontal
barplot(t.stress , horiz = TRUE)


  • Mudar nome das categorías
barplot(t.stress, names.arg = c("A", "B"))


  • Mudar cor interno e bordes
barplot(t.stress, 
        col = "grey", 
        border = "blue") # Uma unica cor


  • Mudar cor dos bordes, por categoria
barplot(t.stress, 
        col = "grey",                       
        border = c("#999999", "#E69F00")) # Cor por categoria


  • Mudar cor interno
barplot(t.stress, col = "steelblue")


  • Mudar cor interno, por categoria
barplot(t.stress, col = c("#999999", "#E69F00"))


  • Modificar titulo (main title) e etiqutas eixos (axis labels)
barplot(t.stress, 
        main = "Stress dos fumadores",
        xlab = "Stress", 
        ylab = "Frequência")
